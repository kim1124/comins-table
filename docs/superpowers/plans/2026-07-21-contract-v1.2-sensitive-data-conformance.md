- [ ] **Step 3: Implement the redacted identity preflight**

Create `scripts/check-public-release-identity.mjs`:

```js
import { execFileSync } from "node:child_process";
import { resolve } from "node:path";
import { pathToFileURL } from "node:url";

const isNoreply = (value) => /@users\.noreply\.github\.com$/i.test(value) || /^noreply@github\.com$/i.test(value);

export function validatePublicIdentity({ owner, gitName, gitEmail, npmEmail }) {
  const valid = owner && gitName.toLowerCase() === owner.toLowerCase() && isNoreply(gitEmail) && isNoreply(npmEmail);
  if (!valid) throw new Error("Public release identity check failed");
}

function command(command, args) {
  try {
    return execFileSync(command, args, { encoding: "utf8", stdio: ["ignore", "pipe", "pipe"] }).trim();
  } catch {
    throw new Error("Public release identity check failed");
  }
}

function run() {
  const remote = command("git", ["remote", "get-url", "origin"]);
  const owner = /github\.com[/:]([^/]+)\//i.exec(remote)?.[1] ?? "";
  const profile = JSON.parse(command("npm", ["profile", "get", "--json"]));
  validatePublicIdentity({
    owner,
    gitName: command("git", ["config", "--get", "user.name"]),
    gitEmail: command("git", ["config", "--get", "user.email"]),
    npmEmail: typeof profile.email === "string" ? profile.email : "",
  });
  process.stdout.write("Public release identity check passed.\n");
}

if (import.meta.url === pathToFileURL(resolve(process.argv[1] ?? "")).href) {
  try {
    run();
  } catch {
    process.stderr.write("Public release identity check failed.\n");
    process.exitCode = 1;
  }
}
```
